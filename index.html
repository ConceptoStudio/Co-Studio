function updateGallery() { const galleryContainer = document.querySelector('.gallery-container'); const projectCards = document.querySelectorAll('.project-card'); // Check screen size before applying transforms if (window.innerWidth >= 1024) { const scrollY = window.scrollY; const translateY = scrollY * 0.3; // Adjust value for parallax effect // Apply transforms to gallery container galleryContainer.style.transform = `translateY(-${translateY}px)`; projectCards.forEach((card, index) => { const cardTranslateY = (scrollY - (index * 100)) * 0.5; // Adjust value based on card index card.style.transform = `translateY(-${cardTranslateY}px)`; }); } else { // Disable transforms on smaller screens galleryContainer.style.transform = 'none'; projectCards.forEach(card => { card.style.transform = 'none'; }); } }
